# PLAN.164 â€” Native Schema Core Threading

## Context

Plan 163 cleared the gate: all 7 type claims pass `tsc --noEmit` and `vitest run`. The
scratch proof also surfaced a design change from the plan 136 spec â€” `ScalarSchema<T>`
(single generic union) was redesigned to specific named concrete types (`StringSchema`,
`LiteralSchema<V>`, `OptionalSchema<S>`, etc.) to avoid `Infer<>` distributing over the
recursive optional member. The scratch design is now the canonical design.

**North star**: remove `zod` entirely from `@arbor/router` so users can bring their own
schema library for user-controlled positions (body, response). Plan 164 removes Zod from
router-controlled positions (path params). Plan 165 (parallel) removes it from body/response.

Current Zod coupling in scope:
- `RouteNode.schema: z.ZodObject<any,any> | null` â†’ `AnyObjectSchema | null`
- `route<S extends z.ZodObject>()` â†’ `route<S extends AnyObjectSchema>()`
- `walk.ts`: `getTag`, `getShape`, `validateSchema` â€” operate on Zod today
- `generate-spec.ts`: `buildQueryParams` â€” reads Zod shape from `node.schema`

**Important constraint**: the Zod value import in `walk.ts` **cannot be fully removed** in
this plan. The `querySchema` field in `_meta` is still `z.ZodObject` until plan 165.
`getShape` and `validateSchema` must remain for that path. The Zod import downgrades from
value to type-only only after plan 165 completes.

## Goal

Create `src/core/schema.ts` with the scratch-proven types + factories + a runtime
`parseObjectSchema` function, then thread `AnyObjectSchema` through `RouteNode`, `route()`,
`walk.ts`, and `generate-spec.ts`. All existing tests pass at every gate.

### Step 4a â€” Create `src/core/schema.ts`

Use the validated design from the scratch proof (plan 163). No imports from other router
modules â€” this file is standalone.

Key design point: `object<F>()` captures the concrete field-schema map `F` and computes
`T = { [K in keyof F]: Infer<F[K]> }` at the call site so `Infer<ObjectSchema<T>>` is
O(1) â€” no per-field recursion at inference time.

```typescript
export type StringConstraints = {
  format?: 'email' | 'url' | 'uuid' | 'cuid' | 'ulid' | 'datetime' | 'date' | 'time';
  minLength?: number;
  maxLength?: number;
  pattern?: string;
};

export type NumberConstraints = {
  minimum?: number;  maximum?: number;
  exclusiveMinimum?: number;  exclusiveMaximum?: number;
  multipleOf?: number;
};

export type StringSchema  = { kind: 'string';   constraints?: StringConstraints };
export type NumberSchema  = { kind: 'number';   constraints?: NumberConstraints };
export type IntegerSchema = { kind: 'integer';  constraints?: NumberConstraints };
export type BooleanSchema = { kind: 'boolean' };
export type LiteralSchema<V extends string | number | boolean> = { kind: 'literal'; value: V };
export type OptionalSchema<S> = { kind: 'optional'; inner: S };
export type BrandSchema<S>    = { kind: 'brand';    inner: S; brand: string };

// Inline object forms for optional/brand avoid the generic-alias circularity
// TypeScript flags on OptionalSchema<AnyScalarSchema>.
export type AnyScalarSchema =
  | StringSchema | NumberSchema | IntegerSchema | BooleanSchema
  | LiteralSchema<string | number | boolean>
  | { kind: 'optional'; inner: AnyScalarSchema }
  | { kind: 'brand'; inner: AnyScalarSchema; brand: string };

export type ObjectSchema<T extends Record<string, unknown>> = {
  kind: 'object';
  fields: { [K in keyof T]: AnyScalarSchema };
};

export type AnyObjectSchema = ObjectSchema<Record<string, unknown>>;

export type Infer<S> =
  S extends ObjectSchema<infer T>             ? T :
  S extends { kind: 'string' }               ? string :
  S extends { kind: 'number' | 'integer' }   ? number :
  S extends { kind: 'boolean' }              ? boolean :
  S extends LiteralSchema<infer V>           ? V :
  S extends OptionalSchema<infer I>          ? Infer<I> | undefined :
  S extends BrandSchema<infer I>             ? Infer<I> :
  never;

// Factories
export const string  = (c?: StringConstraints): StringSchema  => ({ kind: 'string', constraints: c });
export const number  = (c?: NumberConstraints): NumberSchema  => ({ kind: 'number', constraints: c });
export const integer = (c?: NumberConstraints): IntegerSchema => ({ kind: 'integer', constraints: c });
export const boolean = (): BooleanSchema => ({ kind: 'boolean' });
export const literal = <V extends string | number | boolean>(value: V): LiteralSchema<V> => ({ kind: 'literal', value });
export const optional = <S extends AnyScalarSchema>(inner: S): OptionalSchema<S> => ({ kind: 'optional', inner });
export const object = <F extends Record<string, AnyScalarSchema>>(
  fields: F,
): ObjectSchema<{ [K in keyof F]: Infer<F[K]> }> => ({ kind: 'object', fields: fields as never });
export const email = (c?: Omit<StringConstraints, 'format'>): StringSchema => string({ format: 'email', ...c });
export const uuid  = (c?: Omit<StringConstraints, 'format'>): StringSchema => string({ format: 'uuid',  ...c });
export const url   = (c?: Omit<StringConstraints, 'format'>): StringSchema => string({ format: 'url',   ...c });
```

#### Runtime parse function

Path params arrive already coerced by segment matching (`:id` â†’ string, `#id` â†’ number),
so the validator only needs type-checking. Optional fields return `undefined` when absent;
brand schemas are TS-only and delegate to their inner schema at runtime.

```typescript
export type SchemaIssue = { message: string; path?: (string | number)[] };

export function parseObjectSchema(
  schema: AnyObjectSchema,
  input: Record<string, unknown>,
): { success: true; data: Record<string, unknown> } | { success: false; issues: SchemaIssue[] } {
  const data: Record<string, unknown> = {};
  const issues: SchemaIssue[] = [];
  for (const [key, field] of Object.entries(schema.fields)) {
    const result = parseScalar(field as AnyScalarSchema, input[key]);
    if (result.ok) data[key] = result.value;
    else issues.push({ message: result.message, path: [key] });
  }
  return issues.length === 0 ? { success: true, data } : { success: false, issues };
}

type ScalarResult = { ok: true; value: unknown } | { ok: false; message: string };

function parseScalar(schema: AnyScalarSchema, value: unknown): ScalarResult {
  switch (schema.kind) {
    case 'string':  return typeof value === 'string'  ? { ok: true, value } : { ok: false, message: `expected string, got ${typeof value}` };
    case 'number':
    case 'integer': return typeof value === 'number'  ? { ok: true, value } : { ok: false, message: `expected number, got ${typeof value}` };
    case 'boolean': return typeof value === 'boolean' ? { ok: true, value } : { ok: false, message: `expected boolean, got ${typeof value}` };
    case 'literal': return value === schema.value     ? { ok: true, value } : { ok: false, message: `expected ${String(schema.value)}, got ${String(value)}` };
    case 'optional': return value === undefined || value === null ? { ok: true, value: undefined } : parseScalar(schema.inner, value);
    case 'brand':    return parseScalar(schema.inner, value);
  }
}
```

function validateNative(
  schema: AnyObjectSchema,
  value: unknown,
  path: string,
  diag?: ParseDiag[],
): Record<string, unknown> | undefined {
  const input = value != null && typeof value === 'object' ? (value as Record<string, unknown>) : {};

function scalarToJsonSchema(s: AnyScalarSchema): Record<string, unknown> {
  switch (s.kind) {
    case 'string':   return { type: 'string', ...flattenStringConstraints(s.constraints) };
    case 'number':   return { type: 'number', ...s.constraints };
    case 'integer':  return { type: 'integer', ...s.constraints };
    case 'boolean':  return { type: 'boolean' };
    case 'literal':  return { const: s.value };
    case 'optional': return scalarToJsonSchema(s.inner);
    case 'brand':    return scalarToJsonSchema(s.inner);
  }
}

function flattenStringConstraints(c?: StringConstraints): Record<string, unknown> {
  if (!c) return {};
  const out: Record<string, unknown> = {};
  if (c.format)    out['format']    = c.format;
  if (c.minLength !== undefined) out['minLength'] = c.minLength;
  if (c.maxLength !== undefined) out['maxLength'] = c.maxLength;
  if (c.pattern)   out['pattern']   = c.pattern;
  return out;
}
