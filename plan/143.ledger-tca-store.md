# Plan 143 — Ledger TCA Store & Env

## State Shape

```typescript
// ledger.store.ts
import type { DisplayGroupsResponse } from '@arbor/api';  // inferred from route response schema
import type { LedgerEnv } from './ledger.env.js';

export type LedgerLoadState =
  | { tag: 'idle' }
  | { tag: 'loading' }
  | { tag: 'loaded'; groups: DisplayGroupsResponse }
  | { tag: 'error'; message: string };

export interface LedgerState {
  loadState: LedgerLoadState;
  selectedIndex: number;
}

export type LedgerAction =
  | { tag: 'fetch' }
  | { tag: 'loaded'; groups: DisplayGroupsResponse }
  | { tag: 'error'; message: string };

export const initialLedgerState: LedgerState = {
  loadState: { tag: 'idle' },
  selectedIndex: 0,
};
```

## Reducer

```typescript
export const ledgerReducer: Reducer<LedgerState, LedgerAction, LedgerEnv> = ($, action, env) => {
  switch (action.tag) {
    case 'fetch': {
      $.state.loadState = { tag: 'loading' };
      return env.fetchQueue.map((result) =>
        result.fold<LedgerAction>(
          (groups) => ({ tag: 'loaded', groups }),
          (err) => ({ tag: 'error', message: err }),
        ),
      );
    }
    case 'loaded': {
      $.state.loadState = { tag: 'loaded', groups: action.groups };
      return null;
    }
    case 'error': {
      $.state.loadState = { tag: 'error', message: action.message };
      return null;
    }
  }
};
```

## Env

```typescript
// ledger.env.ts
import { createClient } from '@arbor/router';
import { ledgerRouter } from '@arbor/api/ledger';
import { Effect, Result } from '@arbor/common';

export interface LedgerEnv {
  fetchQueue: Effect<Result<DisplayGroupsResponse, string>>;
}

export const liveLedgerEnv: LedgerEnv = {
  fetchQueue: Effect.fromPromise(async () => {
    const client = createClient('http://localhost:3001', ledgerRouter);
    const resp = await client.fetch({ tag: 'ledger-get-queue' });
    if (resp.status === 200) return Result.ok(resp.body);
    return Result.failure('fetch failed');
  }),
};
```

Note: `createClient` without a custom `fetch` uses `globalThis.fetch`. For local dev with
CORS open on the API, this works directly.

## Success Criteria

- `ledgerReducer` handles `fetch → loading → loaded` sequence without TypeScript errors.
- `liveLedgerEnv.fetchQueue` resolves with real data when the API is running.
- `pnpm --filter @arbor/ui typecheck` clean.
